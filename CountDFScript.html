def binary_search(seq, t):
    min = 0
    max = len(seq) - 1
    while True:
        if max < min:
            return -1
        m = (min + max) // 2
        if seq[m] < t:
            min = m + 1
        elif seq[m] > t:
            max = m - 1
        else:
            return m

        print('%d of %d rows covered' %(x,N))
    if (binary_search(Sample,x) != -1 or sampling == 'no'):
        keys = row.split(',')[1:]
        click = keys[0]
        keys[0] = 'weekend' if (keys[1][4:6] in ['25','26']) else 'weekday'
        keys[1] = int(keys[1][6:8])
        for i in range(len(features)):
            if(i != features.index("device_ip")):
                if(not NumRecords[i]):

                if(keys[i] not in NumRecords[i].keys()):
